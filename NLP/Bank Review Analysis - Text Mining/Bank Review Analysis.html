# function to check and get the part of speech tag count of a words in a given sentence
def check_pos_tag(x, flag):
    cnt = 0
    try:
        text=word_tokenize(x)
        wiki = nltk.pos_tag(text)
        for tup in wiki:
            ppo = list(tup)[1]
            if ppo in pos_family[flag]:
                cnt += 1
    except:
        #print('e')
        pass
    return cnt

def create_corpus(x):
    corpus = []
    wn = WordNetLemmatizer()
    ps = PorterStemmer()
    for i in tqdm_notebook(range(0, x.shape[0])):
        #print(i,end='\r')
        msg = re.sub('[^a-zA-Z]', ' ', x[i])
        msg = msg.lower()
        msg = msg.split()
        #msg = word_tokenize(X_train[i].lower()) #Not preferred as it doesn't handle metachars, numbers
        msg = [wn.lemmatize(word) for word in msg if not word in set(eng_stopwords)] #wn.lemmatize #ps.stem
        msg = ' '.join(msg)
        corpus.append(msg)
        #break
    return corpus

# We use textblob utility to calculate polarity
def get_sentiment(x):
    # create TextBlob object of passed text 
    analysis = TextBlob(x) 
    # set sentiment 
    if analysis.sentiment.polarity > 0: 
        return 'positive'
    elif analysis.sentiment.polarity == 0: 
        return 'neutral'
    else: 
        return 'negative'

def pred_rate(cls, X_train, X_test, y_train, y_test):
    # fitting
    cls.fit(X_train, y_train) 
    # pred
    y_pred_train = cls.predict(X_train)
    y_pred_test = cls.predict(X_test)

    print('cm train\n',confusion_matrix(y_train, y_pred_train))
    print('cm test \n',confusion_matrix(y_test, y_pred_test))

    accuracy = accuracy_score(y_train, y_pred_train)
    print("\nAccuracy train : {} %".format(round(accuracy*100,3)))
    accuracy = accuracy_score(y_test, y_pred_test)
    print("Accuracy test  : {} %".format(round(accuracy*100,3)))
    
    train_auc = roc_auc_score(y_train, y_pred_train)
    test_auc  = roc_auc_score(y_test, y_pred_test)
    print("\nAUC Train : ", train_auc)
    print("AUC Test  : ", test_auc)
    
    print('\nClassification Report Train :')
    print(classification_report(y_train, y_pred_train))
    print('\nClassification Report Test :')
    print(classification_report(y_test, y_pred_test))
    print('-----------------------------------------------------------')

def get_top_features_cluster(X_train_tv, prediction, n_feats):
    labels = np.unique(prediction)
    dfs = []
    for label in labels:
        id_temp = np.where(prediction==label) # indices for each cluster
        x_means = np.mean(X_train_tv[id_temp], axis = 0) # returns average score across cluster
        sorted_means = np.argsort(x_means)[::-1][:n_feats] # indices with top 20 scores
        features = tv.get_feature_names()
        best_features = [(features[i], x_means[i]) for i in sorted_means]
        df = pd.DataFrame(best_features, columns = ['features', 'score'])
        dfs.append(df)
    return dfs
